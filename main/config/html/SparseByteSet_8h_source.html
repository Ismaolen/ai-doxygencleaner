10 /***
11  * SparseByteSet
12  *
13  * A special-purpose data structure representing a set of bytes.
14  * May have better performance than std::bitset<256>, depending on workload.
15  *
16  * Operations:
17  * - add(byte)
18  * - contains(byte)
19  * - clear()
20  *
21  * Performance:
22  * - The entire capacity of the set is inline; the set never allocates.
23  * - The constructor zeros only the first two bytes of the object.
24  * - add and contains both run in constant time w.r.t. the size of the set.
25  * Constant time - not amortized constant - and with small constant factor.
26  *
27  * This data structure is ideal for on-stack use.
28  *
29  * Aho, Hopcroft, and Ullman refer to this trick in "The Design and Analysis
30  * of Computer Algorithms" (1974), but the best description is here:
31  * http://research.swtch.com/sparse
32  * http://citeseerx.ist.psu.edu/viewdoc/summary?doi=10.1.1.30.7319
33  */
34 class SparseByteSet {
35  public:
36  // There are this many possible values:
37  static constexpr uint16_t kCapacity = 256;
38 
39  // No init of byte-arrays required!
40  SparseByteSet() : size_(0) {}
41 
42  /***
43  * add(byte)
44  *
45  * O(1), non-amortized.
46  */
47  inline bool add(uint8_t i) {
48  bool r = !contains(i);
49  if (r) {
50  assert(size_ < kCapacity);
51  dense_[size_] = i;
52  sparse_[i] = uint8_t(size_);
53  size_++;
54  }
55  return r;
56  }
57 
58  /***
59  * contains(byte)
60  *
61  * O(1), non-amortized.
62  */
63  inline bool contains(uint8_t i) const FOLLY_DISABLE_MEMORY_SANITIZER {
64  return sparse_[i] < size_ && dense_[sparse_[i]] == i;
65  }
66 
67  /***
68  * clear()
69  *
70  * O(1), non-amortized.
71  */
72  inline void clear() { size_ = 0; }
73 
74  private:
75  uint16_t size_; // can't use uint8_t because it would overflow if all
76  // possible values were inserted.
77  uint8_t sparse_[kCapacity];
78  uint8_t dense_[kCapacity];
79 };
